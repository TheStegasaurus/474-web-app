Harrisonburg High School

Query

  SELECT
    sch_year, subject, avg_score
  FROM sol_test_data
  WHERE TRUE
    -- Harrisonburg High School
    AND div_num = 113
    AND sch_num = 12
    -- all races, genders, etc.
    AND race = 'ALL'
    AND gender = 'ALL'
    AND disabil = 'ALL'
    AND lep = 'ALL'
    AND disadva = 'ALL'
    -- all tests (by subject)
    AND test_name = 'ALL'
    AND test_level = 'ALL'
  ORDER BY sch_year, subject
